Saude Remota 1.0.0
Sistema de monitorización remota de salud basado en el sensor MAX30102. Permite medir ritmo cardíaco en tiempo real y emplea IA para detectar anomalías y generar alertas automáticas. Ideal para aplicaciones de telemedicina en dispositivos embebidos como el ESP32

35 """Controlador SSD1306 con autochequeo de presencia.
36
37 • self.connected -> True si la pantalla respondió.
38 • is_connected() -> Método auxiliar.
39 • Si no hay pantalla, las funciones de dibujo no hacen nada.
40 """
41 def __init__(self, width=128, height=64, i2c=None, addr=SSD1306_I2C_ADDRESS):
42 self.width = width
43 self.height = height
44 self.pages = height // 8
45 self.addr = addr
46
47 # --- I²C ---
48 if i2c is None:
49 self.i2c = I2C(0, scl=Pin(22), sda=Pin(21), freq=400_000)
50 else:
51 self.i2c = i2c
52
53 # --- ¿Pantalla presente? ---
54 self.connected = self._detect_display()
55
56 # Buffer y framebuf se crean siempre; así el código que escribe
57 # texto/dibujos no revienta, aunque luego no se envíe a HW
58 self.buffer = bytearray(self.pages * self.width)
59 self.framebuf = framebuf.FrameBuffer(self.buffer, self.width,
60 self.height, framebuf.MONO_VLSB)
61
62 if self.connected:
63 self._init_display()
64 else:
65 # Convertir llamadas críticas en NOP
66 self.write_cmd = self._noop
67 self.write_data = self._noop
68 self.show = self._noop

151 def text_scaled(self, string, x, y, scale=1, color=1):
152 """Renderiza texto escalado"""
153 for char in string:
154 # Buffer temporal para el caracter
155 char_width = 8
156 char_height = 8
157 buf_temp = bytearray(char_width * char_height // 8)
158 fb_temp = framebuf.FrameBuffer(buf_temp, char_width, char_height, framebuf.MONO_VLSB)
159
160 # Renderizar caracter
161 fb_temp.fill(0)
162 fb_temp.text(char, 0, 0, color)
163
164 # Transferir con escala
165 for py in range(char_height):
166 for px in range(char_width):
167 if fb_temp.pixel(px, py):
168 self.framebuf.fill_rect(
169 x + px * scale,
170 y + py * scale,
171 scale, scale, color)
172
173 x += char_width * scale
