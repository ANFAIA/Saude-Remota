92 def calculate_spo2_and_heart_rate(self, ir_buffer, red_buffer):
93 """Algoritmo completo descrito en AN‑6595; vermente implementa:
94 1. Eliminación de componente DC e inversión de señal IR.
95 2. Suavizado con media móvil de 4 puntos.
96 3. Cálculo de umbral dinámico y detección de valles.
97 4. Estimación de bpm mediante intervalos entre valles.
98 5. Estimación de SpO2 usando relación AC/DC y la tabla ``SPO2_TABLE``.
99 Consulte el código para detalle de cálculos intermedios."""
100 # Verificar que los buffers tengan la misma longitud y sean válidos
101 if len(ir_buffer) != len(red_buffer) or len(ir_buffer) < 4:
102 return -999, 0, -999, 0
103
104 # 1. Calcula la media DC y elimina DC de IR, invierte señal
105 un_ir_mean = self._mean(ir_buffer)
106 an_x = [-1 * (val - un_ir_mean) for val in ir_buffer]
107
108 # 2. Media móvil de 4 puntos - CORRECCIÓN: usar longitud real del buffer
109 an_x_ma4 = an_x.copy()
110 buffer_length = len(an_x)
111 for k in range(buffer_length - self.MA4_SIZE):
112 an_x_ma4[k] = sum(an_x[k:k+self.MA4_SIZE]) // self.MA4_SIZE
113
114 # 3. Calcula umbral
115 n_th1 = int(self._mean(an_x_ma4[:buffer_length - self.MA4_SIZE]))
116 n_th1 = max(30, min(n_th1, 60))
117
118 # 4. Detección de valles (picos en señal invertida)
119 # mínimo 160 ms entre valles ⇒ muestras = FreqS * 0.16
120 #min_gap = self.FreqS // 6 # ≈0.166 s
121 #an_ir_valley_locs = self._find_peaks(an_x_ma4, n_th1, min_gap, 15)
122 an_ir_valley_locs = self._find_peaks(an_x_ma4, n_th1, 4, 15)
123
124 n_npks = len(an_ir_valley_locs)
125
126 # 5. Heart rate
127 if n_npks >= 2:
128 n_peak_interval_sum = 0
129 for k in range(1, n_npks):
130 # Asegurar que los índices estén dentro del rango
131 if (an_ir_valley_locs[k] < buffer_length and
132 an_ir_valley_locs[k-1] < buffer_length):
133 n_peak_interval_sum += (an_ir_valley_locs[k] - an_ir_valley_locs[k-1])
134 if n_peak_interval_sum > 0:
135 n_peak_interval_sum //= (n_npks - 1)
136 heart_rate = int((self.FreqS * 60) / n_peak_interval_sum)
137 hr_valid = 1
138 else:
139 heart_rate = -999
140 hr_valid = 0
141 else:
142 heart_rate = -999
143 hr_valid = 0
144
145 # 6. Carga valores originales para SpO2
146 an_x = list(ir_buffer)
147 an_y = list(red_buffer)
148
149 # 7. Calcula SpO2 usando los valles detectados
150 n_exact_ir_valley_locs_count = n_npks
151 an_ratio = []
152 for k in range(n_exact_ir_valley_locs_count - 1):
153 loc_k = an_ir_valley_locs[k]
154 loc_k1 = an_ir_valley_locs[k+1]
155
156 # Verificar que los índices estén dentro del rango CORRECCIÓN PRINCIPAL
157 if (loc_k < buffer_length and loc_k1 < buffer_length and
158 loc_k1 > loc_k and loc_k1 - loc_k > 3):
159
160 # Busca máximos DC en el segmento
161 segment_end = min(loc_k1 + 1, buffer_length)
162 segment_x = an_x[loc_k:segment_end]
163 segment_y = an_y[loc_k:segment_end]
164
165 if segment_x and segment_y:
166 n_x_dc_max, n_x_dc_max_idx_rel = self._max(segment_x)
167 n_x_dc_max_idx = loc_k + n_x_dc_max_idx_rel
168 n_y_dc_max, n_y_dc_max_idx_rel = self._max(segment_y)
169 n_y_dc_max_idx = loc_k + n_y_dc_max_idx_rel
170
171 # AC componente IR y RED
172 # Cálculo seguro con verificación de índices
173 if loc_k1 < buffer_length:
174 n_y_ac = (an_y[loc_k1] - an_y[loc_k]) * (n_y_dc_max_idx - loc_k)
175 n_y_ac = an_y[loc_k] + n_y_ac // (loc_k1 - loc_k)
176 n_y_ac = an_y[n_y_dc_max_idx] - n_y_ac
177
178 n_x_ac = (an_x[loc_k1] - an_x[loc_k]) * (n_x_dc_max_idx - loc_k)
179 n_x_ac = an_x[loc_k] + n_x_ac // (loc_k1 - loc_k)
180 n_x_ac = an_x[n_x_dc_max_idx] - n_x_ac
181
182 n_nume = (n_y_ac * n_x_dc_max) >> 7
183 n_denom = (n_x_ac * n_y_dc_max) >> 7
184
185 if n_denom > 0 and n_nume != 0:
186 ratio = (n_nume * 100) // n_denom
187 an_ratio.append(ratio)
188
189 # 8. Mediana de ratios
190 if len(an_ratio) > 0:
191 an_ratio.sort()
192 n_middle_idx = len(an_ratio) // 2
193 if n_middle_idx > 0:
194 n_ratio_average = (an_ratio[n_middle_idx-1] + an_ratio[n_middle_idx]) // 2
195 else:
196 n_ratio_average = an_ratio[n_middle_idx]
197 else:
198 n_ratio_average = -1
199
200 # 9. Busca SpO2 en tabla
201 if 2 < n_ratio_average < len(self.SPO2_TABLE):
202 spo2 = self.SPO2_TABLE[int(n_ratio_average)]
203 spo2_valid = 1
204 else:
205 spo2 = -999
206 spo2_valid = 0
207
208 return spo2, spo2_valid, heart_rate, hr_valid

215 def _find_peaks(self, x, min_height, min_distance, max_num):
216 """
217 Encuentra hasta max_num picos en x mayores que min_height y separados al menos min_distance.
218 Devuelve lista de índices.
219 """
220 peaks = []
221 i = 1
222 n_size = len(x)
223 while i < n_size - 1:
224 if x[i] > min_height and x[i] > x[i-1]:
225 width = 1
226 # Evitar desbordamiento en el bucle while
227 while (i + width < n_size) and (x[i] == x[i+width]):
228 width += 1
229 if (i + width < n_size) and (x[i] > x[i+width]):
230 peaks.append(i)
231 i += width + 1
232 else:
233 i += width
234 else:
235 i += 1
236 # Elimina picos demasiado cercanos
237 peaks = self._remove_close_peaks(x, peaks, min_distance)
238 # Limita a max_num
239 peaks = peaks[:max_num]
240 return peaks

247 def _remove_close_peaks(self, x, peaks, min_distance):
248 """
249 Ordena los picos por altura descendente y elimina los que estén demasiado cerca.
250 """
251 if not peaks:
252 return []
253 # Ordena por valor descendente
254 peaks_sorted = sorted(peaks, key=lambda idx: x[idx], reverse=True)
255 filtered = []
256 for idx in peaks_sorted:
257 if all(abs(idx - fidx) > min_distance for fidx in filtered):
258 filtered.append(idx)
259 # Devuelve ordenados por posición ascendente
260 return sorted(filtered)
