462 def setup(self, powerLevel, sampleAverage, ledMode, sampleRate, pulseWidth, adcRange):
463 self.softReset()
464 # FIFO Avg
465 samp_map = {1:MAX30105_SAMPLEAVG_1,2:MAX30105_SAMPLEAVG_2,4:MAX30105_SAMPLEAVG_4,
466 8:MAX30105_SAMPLEAVG_8,16:MAX30105_SAMPLEAVG_16,32:MAX30105_SAMPLEAVG_32}
467 self.setFIFOAverage(samp_map.get(sampleAverage, MAX30105_SAMPLEAVG_4))
468 self.enableFIFORollover()
469 # Mode
470 modes = {1:MAX30105_MODE_REDONLY,2:MAX30105_MODE_REDIRONLY,3:MAX30105_MODE_MULTILED}
471 self.setLEDMode(modes.get(ledMode, MAX30105_MODE_REDONLY))
472 self.activeLEDs = ledMode
473 # ADC range
474 if adcRange<4096: self.setADCRange(MAX30105_ADCRANGE_2048)
475 elif adcRange<8192: self.setADCRange(MAX30105_ADCRANGE_4096)
476 elif adcRange<16384: self.setADCRange(MAX30105_ADCRANGE_8192)
477 elif adcRange==16384: self.setADCRange(MAX30105_ADCRANGE_16384)
478 else: self.setADCRange(MAX30105_ADCRANGE_2048)
479 # Sample rate
480 rate_map = [(100,MAX30105_SAMPLERATE_50),
481 (200,MAX30105_SAMPLERATE_100),
482 (400,MAX30105_SAMPLERATE_200),
483 (800,MAX30105_SAMPLERATE_400),
484 (1000,MAX30105_SAMPLERATE_800),
485 (1600,MAX30105_SAMPLERATE_1000),
486 (3200,MAX30105_SAMPLERATE_1600)]
487 for thresh,code in rate_map:
488 if sampleRate < thresh:
489 self.setSampleRate(code)
490 break
491 else:
492 self.setSampleRate(MAX30105_SAMPLERATE_3200 if sampleRate==3200 else MAX30105_SAMPLERATE_50)
493 # Pulse width
494 pw_map = [(118,MAX30105_PULSEWIDTH_69),
495 (215,MAX30105_PULSEWIDTH_118),
496 (411,MAX30105_PULSEWIDTH_215)]
497 for thresh,code in pw_map:
498 if pulseWidth < thresh:
499 self.setPulseWidth(code)
500 break
501 else:
502 self.setPulseWidth(MAX30105_PULSEWIDTH_411 if pulseWidth==411 else MAX30105_PULSEWIDTH_69)
503 # LED amplitudes
504 for fn in (self.setPulseAmplitudeRed,
505 self.setPulseAmplitudeIR,
506 self.setPulseAmplitudeGreen,
507 self.setPulseAmplitudeProximity):
508 fn(powerLevel)
509 # Multi-LED slots
510 self.enableSlot(1, SLOT_RED_LED)
511 if ledMode>1: self.enableSlot(2, SLOT_IR_LED)
512 if ledMode>2: self.enableSlot(3, SLOT_GREEN_LED)
513 self.clearFIFO()

582 def check(self):
583 readPtr = self.getReadPointer()
584 writePtr = self.getWritePointer()
585 if readPtr == writePtr:
586 return 0
587 num = writePtr - readPtr
588 if num < 0: num += STORAGE_SIZE
589 to_read = num * self.activeLEDs * 3
590
591 # burst read
592 while to_read > 0:
593 chunk = min(to_read, 32)
594 # align chunk to multiple of leds*3
595 mod = chunk % (self.activeLEDs*3)
596 if mod:
597 chunk -= mod
598 if chunk == 0: # no hay suficientes datos para leer
599 break
600 try:
601 buf = self.i2c.readfrom_mem(self.addr, MAX30105_FIFODATA, chunk)
602 except OSError:
603 return 0 # Error en I2C
604
605 i = 0
606 while i < len(buf):
607 # Verificar que tenemos suficientes bytes para al menos una muestra
608 if i + 3 > len(buf):
609 break
610
611 self.head = (self.head + 1) % STORAGE_SIZE
612
613 # Leer valor RED (siempre presente)
614 val = (buf[i] << 16) | (buf[i+1] << 8) | buf[i+2]
615 val &= 0x3FFFF
616 self.red[self.head] = val
617 i += 3
618
619 # Leer valor IR (si hay más de 1 LED activo)
620 if self.activeLEDs > 1:
621 if i + 3 > len(buf):
622 break
623 val = (buf[i] << 16) | (buf[i+1] << 8) | buf[i+2]
624 val &= 0x3FFFF
625 self.IR[self.head] = val
626 i += 3
627
628 # Leer valor GREEN (si hay más de 2 LEDs activos)
629 if self.activeLEDs > 2:
630 if i + 3 > len(buf):
631 break
632 val = (buf[i] << 16) | (buf[i+1] << 8) | buf[i+2]
633 val &= 0x3FFFF
634 self.green[self.head] = val
635 i += 3
636
637 to_read -= chunk
638 return num
